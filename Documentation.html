Complete API reference for bhavyTr

createTensor

Tensor createTensor(int* shape, int ndims);

Allocates memory and initializes tensor structure.

Parameters: shape (array), ndims (dimensions)

Returns: Tensor

Time: O(n)

int shape[2]={2,2};
Tensor t=createTensor(shape,2);

setTensorData

void setTensorData(Tensor* t,int* values,int size);

Copies user data into tensor memory.

Constraint: size must match total elements

int vals[4]={1,2,3,4};
setTensorData(&t,vals,4);

getTensorIdx

int getTensorIdx(Tensor t,int* indices);

Maps multi-dimensional index to flat memory.

Time: O(d)

int idx[2]={1,1};
int flat=getTensorIdx(t,idx);
int val=t.data[flat];

addTensor

Tensor addTensor(Tensor a,Tensor b);

Element-wise addition.

Constraint: shapes must match

Time: O(n)

// [1,2,3,4] + [5,6,7,8]
// = [6,8,10,12]

subTensor

Tensor subTensor(Tensor a,Tensor b);

Element-wise subtraction.

// [1,2,3,4] - [5,6,7,8]
// = [-4,-4,-4,-4]

mulTensor

Tensor mulTensor(Tensor a,Tensor b);

Element-wise multiplication.

// [1,2,3,4] * [5,6,7,8]
// = [5,12,21,32]

scalarMultiply

void scalarMultiply(Tensor* t,int scalar);

In-place scalar multiplication.

// [1,2,3,4] * 2
// = [2,4,6,8]

reshapeTensor

Tensor reshapeTensor(Tensor t,int* newShape,int newDims);

Changes shape without changing data.

Constraint: total elements must match

// 2x2 -> 1D
// [2,4,6,8]

freeTensor

void freeTensor(Tensor* t);

Releases memory.

Complete End-to-End Example

Create → set data → index → operations → reshape → free

#include "tensor.h"

int main() {
  int shape[2] = {2, 2};

  // Create tensors
  Tensor a = createTensor(shape, 2);
  Tensor b = createTensor(shape, 2);

  // Set data
  int valsA[4] = {1, 2, 3, 4};
  int valsB[4] = {5, 6, 7, 8};
  setTensorData(&a, valsA, 4);
  setTensorData(&b, valsB, 4);

  // Index access
  int idx[2] = {1, 1};
  int flat = getTensorIdx(a, idx);
  int value = a.data[flat]; // 4

  // Operations
  Tensor c = addTensor(a, b); // [6,8,10,12]
  Tensor d = subTensor(a, b); // [-4,-4,-4,-4]
  Tensor e = mulTensor(a, b); // [5,12,21,32]

  // In-place scalar
  scalarMultiply(&a, 2); // [2,4,6,8]

  // Reshape
  int newShape[1] = {4};
  Tensor f = reshapeTensor(a, newShape, 1); // [2,4,6,8]

  // Free memory
  freeTensor(&a);
  freeTensor(&b);
  freeTensor(&c);
  freeTensor(&d);
  freeTensor(&e);
  freeTensor(&f);

  return 0;
}
